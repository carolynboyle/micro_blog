# .devcontainer/Dockerfile
FROM python:3.11-slim

# Install system dependencies
RUN apt-get update && apt-get install -y --no-install-recommends \
    git \
    build-essential \
    sudo \
    curl \
    && rm -rf /var/lib/apt/lists/*

# Create non-root user with configurable UID/GID for cross-platform compatibility
ARG USER_UID=1000
ARG USER_GID=1000
RUN groupadd --gid $USER_GID devuser \
    && useradd --uid $USER_UID --gid $USER_GID --shell /bin/bash --create-home devuser \
    && echo devuser ALL=\(root\) NOPASSWD:ALL > /etc/sudoers.d/devuser \
    && chmod 0440 /etc/sudoers.d/devuser

# Set up Python environment
RUN pip install --no-cache-dir --upgrade pip

# Create workspace and set ownership
WORKDIR /workspace
RUN chown -R devuser:devuser /workspace

# Switch to non-root user
USER devuser

# Add user's local bin to PATH for pip installed packages
ENV PATH="/home/devuser/.local/bin:$PATH"

# Set Flask environment variables
ENV FLASK_APP=microblog.app
ENV FLASK_ENV=development
ENV FLASK_RUN_HOST=0.0.0.0
ENV FLASK_RUN_PORT=5000
ENV PYTHONPATH=/workspace

# Expose Flask port
EXPOSE 5000

# Default command - will be overridden by docker-compose
CMD ["bash"]
